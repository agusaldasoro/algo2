\subsection{Algoritmos}

	\begin{algorithm}{$i$CrearPromesa}{\param{in}{nT}{nombre}, \param{in}{tP}{tipoPromesa}, \param{in}{d}{dinero}, \param{in}{n}{nat}}{\param{}{res}{estr}}
	res.titulo \leftarrow nT\\
	res.tipo \leftarrow tP \\
	res.limite \leftarrow d \\
	res.cantidad \leftarrow n \\
	\RETURN res
	\end{algorithm}
	
	\begin{algorithm}{$i$Titulo}{\param{in}{p}{estr}}{\param{}{res}{Nombre}}
	res \leftarrow p.titulo \\
	\RETURN res
	\end{algorithm}

	\begin{algorithm}{$i$Tipo}{\param{in}{p}{estr}}{\param{}{res}{TipoPromesa}}
	res \leftarrow p.tipo \\
	\RETURN res
	\end{algorithm}
	
	\begin{algorithm}{$i$Limite}{\param{in}{p}{estr}}{\param{}{res}{Dinero}}
	res \leftarrow p.limite \\
	\RETURN res
	\end{algorithm}
	
	\begin{algorithm}{$i$Cantidad}{\param{in}{p}{estr}}{\param{}{res}{Nat}}
	res \leftarrow p.cantidad \\
	\RETURN res
	\end{algorithm}

	\begin{algorithm}{$i$CopiarPromesa}{\param{in}{p}{estr}}{\param{}{res}{estr}}
	res.titulo \leftarrow Copiar(p.titulo)\\
	res.tipo \leftarrow p.tipo \\
	res.limite \leftarrow p.limite \\
	res.cantidad \leftarrow p.cantidad \\
	\RETURN res
	\end{algorithm}
